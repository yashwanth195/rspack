import { value, b_value } from "./ab-chunk.mjs";

// ./index.js



it('should handle conflicting exports from multi-module chunk', async () => {
	expect(value).toBe(1)
	expect(b_value).toBe(2)

	const modA = await import("./ab-chunk.mjs").then(m => m.a_namespaceObject)
	const modB = await import("./ab-chunk.mjs").then(m => m.b_namespaceObject)

	expect(modA.value).toBe(1)
	expect(modB.value).toBe(2)
})

export {};
