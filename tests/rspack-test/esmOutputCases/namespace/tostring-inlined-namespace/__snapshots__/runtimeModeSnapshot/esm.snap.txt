import { __rspack_context } from "./runtime.mjs";
// NAMESPACE OBJECT: ./foo.js
var foo_namespaceObject = {};
__rspack_context.N(foo_namespaceObject);
__rspack_context.d(foo_namespaceObject, { 
  bar: () => (bar) });


// ./foo.js
const bar = 42;

// ./index.js


/* export default */ const index = ({
  tag: foo_namespaceObject[Symbol.toStringTag],
  text: Object.prototype.toString.call(foo_namespaceObject),
  bar: bar,
});

it("should preserve module toStringTag on local namespace objects", async () => {
  const mod = await import(/* webpackIgnore: true */ "./main.mjs");

  expect(mod.default).toEqual({
    tag: "Module",
    text: "[object Module]",
    bar: 42,
  });
});

export default index;
